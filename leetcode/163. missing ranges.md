# 163 Missing Ranges

```go
// idiomatic, beats 100%
func findMissingRanges(nums []int, lower int, upper int) [][]int {
	if len(nums) == 0 {
		return [][]int{{lower, upper}}
	}
	if lower == upper {
		return nil
	}
	var out [][]int
	start := lower
	for _, num := range nums {
		if num > start {
			out = append(out, []int{start, num - 1})
		}
		start = num + 1
	}
	if upper >= start {
		out = append(out, []int{start, upper})
	}
	return out
}
```

```go
// beats 100%
func findMissingRanges(nums []int, lower int, upper int) [][]int {
    if len(nums) == 0 {
        return [][]int{{lower, upper}}
    }
    if lower == upper {
        return nil
    }
    var out [][]int
    if nums[0] > lower {
        out = append(out, []int{lower, nums[0]-1})
    }
    for i := 1; i < len(nums); i += 1 {
        start := nums[i-1] + 1
        end := nums[i] - 1
        if start <= end {
            out = append(out, []int{start, end})
        }
    }
    if upper > nums[len(nums) - 1] {
        out = append(out, []int{nums[len(nums) - 1] + 1, upper})
    }

    return out
}
```

```javascript
/** slow, beats 68%
 * @param {number[]} nums
 * @param {number} lower
 * @param {number} upper
 * @return {number[][]}
 */
var findMissingRanges = function (nums, lower, upper) {
    if (nums.length === 0) {
        return [[lower, upper]];
    }
    if (lower === upper) {
        return [];
    }
    const lowest = nums[0];
    const uppest = nums[nums.length - 1];
    const out = [];
    if (lowest > lower) {
        out.push([lower, lowest - 1]);
    }
    let start = undefined,
        end = undefined;
    for (let i = 1; i < nums.length; i += 1) {
        start = nums[i - 1] + 1;
        end = nums[i] - 1;
        if (start <= end) {
            out.push([start, end]);
        }
    }
    if (upper > uppest) {
        out.push([uppest + 1, upper]);
    }
    return out;
};
```
